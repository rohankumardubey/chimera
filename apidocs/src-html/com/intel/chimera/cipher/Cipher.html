018package com.intel.chimera.cipher;
019
020import java.io.Closeable;
021import java.nio.ByteBuffer;
022import java.security.InvalidAlgorithmParameterException;
023import java.security.InvalidKeyException;
024import java.util.Properties;
025
026import javax.crypto.BadPaddingException;
027import javax.crypto.IllegalBlockSizeException;
028import javax.crypto.ShortBufferException;
029
030/**
031 * The interface of cryptographic cipher for encryption and decryption.
032 */
033public interface Cipher extends Closeable {
034
035  /**
036   * A constant representing encrypt mode.  The mode constant to be used
037   * when calling init method of the Cipher.
038   */
039  int ENCRYPT_MODE = 1;
040
041  /**
042   * A constant representing decrypt mode.  The mode constant to be used 
043   * when calling init method of the Cipher.
044   */
045  int DECRYPT_MODE = 0;
046
047  /**
048   * Gets the CipherTransformation for this cipher.
049   *
050   * @return the CipherTransformation for this cipher.
051   */
052  CipherTransformation getTransformation();
053
054  /**
055   * Gets the properties for this cipher.
056   * 
057   * @return the properties for this cipher.
058   */
059  Properties getProperties();
060
061  /**
062   * Initializes the cipher with mode, key and iv.
063   * 
064   * @param mode {@link #ENCRYPT_MODE} or {@link #DECRYPT_MODE}
065   * @param key crypto key for the cipher
066   * @param iv Initialization vector for the cipher
067   * @throws InvalidKeyException if the given key is inappropriate for
068   * initializing this cipher, or its keysize exceeds the maximum allowable
069   * keysize (as determined from the configured jurisdiction policy files).
070   * @throws InvalidAlgorithmParameterException if the given algorithm
071   * parameters are inappropriate for this cipher, or this cipher requires
072   * algorithm parameters and <code>params</code> is null, or the given
073   * algorithm parameters imply a cryptographic strength that would exceed
074   * the legal limits (as determined from the configured jurisdiction
075   * policy files).
076   */
077  void init(int mode, byte[] key, byte[] iv)
078      throws InvalidKeyException, InvalidAlgorithmParameterException;
079
080  /**
081   * Continues a multiple-part encryption/decryption operation. The data
082   * is encrypted or decrypted, depending on how this cipher was initialized.
083   * 
084   * @param inBuffer the input ByteBuffer
085   * @param outBuffer the output ByteBuffer
086   * @return int number of bytes stored in <code>output</code>
087   * @throws ShortBufferException if there is insufficient space
088   * in the output buffer
089   */
090  int update(ByteBuffer inBuffer, ByteBuffer outBuffer)
091      throws ShortBufferException;
092
093  /**
094   * Continues a multiple-part encryption/decryption operation. The data
095   * is encrypted or decrypted, depending on how this cipher was initialized.
096   *
097   * @param input the input byte array
098   * @param inputOffset the offset in input where the input starts
099   * @param inputLen the input length
100   * @param output the byte array for the result
101   * @param outputOffset the offset in output where the result is stored
102   * @return the number of bytes stored in output
103   * @throws ShortBufferException if there is insufficient space in the output byte array
104   */
105  int update(byte[] input, int inputOffset, int inputLen,
106      byte[] output, int outputOffset)
107      throws ShortBufferException;
108
109  /**
110   * Encrypts or decrypts data in a single-part operation, or finishes a
111   * multiple-part operation.
112   * 
113   * @param inBuffer the input ByteBuffer
114   * @param outBuffer the output ByteBuffer
115   * @return int number of bytes stored in <code>output</code>
116   * @throws BadPaddingException if this cipher is in decryption mode,
117   * and (un)padding has been requested, but the decrypted data is not
118   * bounded by the appropriate padding bytes
119   * @throws IllegalBlockSizeException if this cipher is a block cipher,
120   * no padding has been requested (only in encryption mode), and the total
121   * input length of the data processed by this cipher is not a multiple of
122   * block size; or if this encryption algorithm is unable to
123   * process the input data provided.
124   * @throws ShortBufferException if the given output buffer is too small
125   * to hold the result
126   */
127  int doFinal(ByteBuffer inBuffer, ByteBuffer outBuffer)
128      throws ShortBufferException, IllegalBlockSizeException,
129      BadPaddingException;
130
131  /**
132   * Encrypts or decrypts data in a single-part operation, or finishes a
133   * multiple-part operation.
134   *
135   * @param input the input byte array
136   * @param inputOffset the offset in input where the input starts
137   * @param inputLen the input length
138   * @param output the byte array for the result
139   * @param outputOffset the offset in output where the result is stored
140   * @return the number of bytes stored in output
141   * @throws ShortBufferException if the given output byte array is too small
142   * to hold the result
143   * @throws BadPaddingException if this cipher is in decryption mode,
144   * and (un)padding has been requested, but the decrypted data is not
145   * bounded by the appropriate padding bytes
146   * @throws IllegalBlockSizeException if this cipher is a block cipher,
147   * no padding has been requested (only in encryption mode), and the total
148   * input length of the data processed by this cipher is not a multiple of
149   * block size; or if this encryption algorithm is unable to
150   * process the input data provided.
151   */
152  int doFinal(byte[] input, int inputOffset, int inputLen,
153      byte[] output, int outputOffset)
154      throws ShortBufferException, IllegalBlockSizeException, BadPaddingException;
155}
