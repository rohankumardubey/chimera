018package com.intel.chimera.cipher;
019
020import java.nio.ByteBuffer;
021import java.security.GeneralSecurityException;
022import java.security.InvalidAlgorithmParameterException;
023import java.security.InvalidKeyException;
024import java.util.Properties;
025
026import javax.crypto.BadPaddingException;
027import javax.crypto.IllegalBlockSizeException;
028import javax.crypto.ShortBufferException;
029import javax.crypto.spec.IvParameterSpec;
030import javax.crypto.spec.SecretKeySpec;
031
032import com.intel.chimera.utils.Utils;
033
034/**
035 * Implements the {@link com.intel.chimera.cipher.Cipher} using JCE provider.
036 */
037public class JceCipher implements Cipher {
038  private final Properties props;
039  private final CipherTransformation transformation;
040  private final javax.crypto.Cipher cipher;
041
042  /**
043   * Constructs a {@link com.intel.chimera.cipher.Cipher} based on JCE
044   * Cipher {@link javax.crypto.Cipher}.
045   * @param props properties for JCE cipher
046   * @param transformation transformation for JCE cipher
047   * @throws GeneralSecurityException if JCE cipher initialize failed
048   */
049  public JceCipher(Properties props, CipherTransformation transformation)
050      throws GeneralSecurityException {
051    this.props = props;
052    this.transformation = transformation;
053
054    String provider = Utils.getJCEProvider(props);
055    if (provider == null || provider.isEmpty()) {
056      cipher = javax.crypto.Cipher.getInstance(transformation.getName());
057    } else {
058      cipher = javax.crypto.Cipher.getInstance(transformation.getName(), provider);
059    }
060  }
061
062  /**
063   * Gets the CipherTransformation for the jce cipher.
064   *
065   * @return the CipherTransformation for this cipher
066   */
067  @Override
068  public CipherTransformation getTransformation() {
069    return transformation;
070  }
071
072  /**
073   * Gets the properties for the jce cipher.
074   *
075   * @return the properties for this cipher.
076   */
077  @Override
078  public Properties getProperties() {
079    return props;
080  }
081
082  /**
083   * Initializes the cipher with mode, key and iv.
084   * 
085   * @param mode {@link #ENCRYPT_MODE} or {@link #DECRYPT_MODE}
086   * @param key crypto key for the cipher
087   * @param iv Initialization vector for the cipher
088   * @throws InvalidAlgorithmParameterException if the given algorithm
089   * parameters are inappropriate for this cipher, or this cipher requires
090   * algorithm parameters and <code>params</code> is null, or the given
091   * algorithm parameters imply a cryptographic strength that would exceed
092   * the legal limits (as determined from the configured jurisdiction
093   * policy files).
094   */
095  @Override
096  public void init(int mode, byte[] key, byte[] iv)
097      throws InvalidKeyException, InvalidAlgorithmParameterException {
098    Utils.checkNotNull(key);
099    Utils.checkNotNull(iv);
100
101    int cipherMode = javax.crypto.Cipher.DECRYPT_MODE;
102    if (mode == ENCRYPT_MODE)
103      cipherMode = javax.crypto.Cipher.ENCRYPT_MODE;
104
105    cipher.init(cipherMode, new SecretKeySpec(key, "AES"),
106        new IvParameterSpec(iv));
107  }
108
109  /**
110   * Continues a multiple-part encryption/decryption operation. The data
111   * is encrypted or decrypted, depending on how this cipher was initialized.
112   * 
113   * @param inBuffer the input ByteBuffer
114   * @param outBuffer the output ByteBuffer
115   * @return int number of bytes stored in <code>output</code>
116   * @throws ShortBufferException if there is insufficient space
117   * in the output buffer
118   */
119  @Override
120  public int update(ByteBuffer inBuffer, ByteBuffer outBuffer)
121      throws ShortBufferException {
122    return cipher.update(inBuffer, outBuffer);
123  }
124
125  /**
126   * Continues a multiple-part encryption/decryption operation. The data
127   * is encrypted or decrypted, depending on how this cipher was initialized.
128   *
129   * @param input the input byte array
130   * @param inputOffset the offset in input where the input starts
131   * @param inputLen the input length
132   * @param output the byte array for the result
133   * @param outputOffset the offset in output where the result is stored
134   * @return the number of bytes stored in output
135   * @throws ShortBufferException if there is insufficient space in the output byte array
136   */
137  @Override
138  public int update(byte[] input, int inputOffset, int inputLen,
139      byte[] output, int outputOffset)
140      throws ShortBufferException {
141    return cipher.update(input, inputOffset, inputLen,
142                         output, outputOffset);
143  }
144
145  /**
146   * Encrypts or decrypts data in a single-part operation, or finishes a
147   * multiple-part operation. The data is encrypted or decrypted, depending
148   * on how this cipher was initialized.
149   *
150   * @param inBuffer the input ByteBuffer
151   * @param outBuffer the output ByteBuffer
152   * @return int number of bytes stored in <code>output</code>
153   * @throws BadPaddingException if this cipher is in decryption mode,
154   * and (un)padding has been requested, but the decrypted data is not
155   * bounded by the appropriate padding bytes
156   * @throws IllegalBlockSizeException if this cipher is a block cipher,
157   * no padding has been requested (only in encryption mode), and the total
158   * input length of the data processed by this cipher is not a multiple of
159   * block size; or if this encryption algorithm is unable to
160   * process the input data provided.
161   * @throws ShortBufferException if the given output buffer is too small
162   * to hold the result
163   */
164  @Override
165  public int doFinal(ByteBuffer inBuffer, ByteBuffer outBuffer)
166      throws ShortBufferException, IllegalBlockSizeException,
167      BadPaddingException {
168    return cipher.doFinal(inBuffer, outBuffer);
169  }
170
171  /**
172   * Encrypts or decrypts data in a single-part operation, or finishes a
173   * multiple-part operation.
174   *
175   * @param input the input byte array
176   * @param inputOffset the offset in input where the input starts
177   * @param inputLen the input length
178   * @param output the byte array for the result
179   * @param outputOffset the offset in output where the result is stored
180   * @return the number of bytes stored in output
181   * @throws ShortBufferException if the given output byte array is too small
182   * to hold the result
183   * @throws BadPaddingException if this cipher is in decryption mode,
184   * and (un)padding has been requested, but the decrypted data is not
185   * bounded by the appropriate padding bytes
186   * @throws IllegalBlockSizeException if this cipher is a block cipher,
187   * no padding has been requested (only in encryption mode), and the total
188   * input length of the data processed by this cipher is not a multiple of
189   * block size; or if this encryption algorithm is unable to
190   * process the input data provided.
191   */
192  @Override
193  public int doFinal(byte[] input, int inputOffset, int inputLen,
194      byte[] output, int outputOffset)
195      throws ShortBufferException, IllegalBlockSizeException, BadPaddingException {
196    return cipher.doFinal(input, inputOffset, inputLen,
197                          output, outputOffset);
198  }
199
200  /**
201   * Closes Jce cipher.
202   */
203  @Override
204  public void close() {
205    // Do nothing
206  }
207}
