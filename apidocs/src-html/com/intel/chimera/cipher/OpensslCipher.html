018package com.intel.chimera.cipher;
019
020import java.io.IOException;
021import java.nio.ByteBuffer;
022import java.security.GeneralSecurityException;
023import java.util.Properties;
024
025import javax.crypto.BadPaddingException;
026import javax.crypto.IllegalBlockSizeException;
027import javax.crypto.ShortBufferException;
028
029import com.intel.chimera.utils.Utils;
030
031/**
032 * Implements the Cipher using JNI into OpenSSL.
033 */
034public class OpensslCipher implements Cipher {
035  private final Properties props;
036  private final CipherTransformation transformation;
037  private final Openssl cipher;
038
039  /**
040   * Constructs a {@link com.intel.chimera.cipher.Cipher} using JNI into OpenSSL
041   * 
042   * @param props properties for OpenSSL cipher
043   * @param transformation transformation for OpenSSL cipher
044   * @throws GeneralSecurityException if OpenSSL cipher initialize failed
045   */
046  public OpensslCipher(Properties props, CipherTransformation transformation)
047      throws GeneralSecurityException {
048    this.props = props;
049    this.transformation = transformation;
050
051    String loadingFailureReason = Openssl.getLoadingFailureReason();
052    if (loadingFailureReason != null) {
053      throw new RuntimeException(loadingFailureReason);
054    }
055
056    cipher = Openssl.getInstance(transformation.getName());
057  }
058
059  /**
060   * Gets the CipherTransformation for the openssl cipher.
061   *
062   * @return the CipherTransformation for this cipher
063   */
064  @Override
065  public CipherTransformation getTransformation() {
066    return transformation;
067  }
068
069  /**
070   * Gets the properties for the openssl cipher.
071   *
072   * @return the properties for this cipher.
073   */
074  @Override
075  public Properties getProperties() {
076    return props;
077  }
078
079  /**
080   * Initializes the cipher with mode, key and iv.
081   * @param mode {@link #ENCRYPT_MODE} or {@link #DECRYPT_MODE}
082   * @param key crypto key for the cipher
083   * @param iv Initialization vector for the cipher
084   * @throws IOException if cipher initialize fails
085   */
086  @Override
087  public void init(int mode, byte[] key, byte[] iv) {
088    Utils.checkNotNull(key);
089    Utils.checkNotNull(iv);
090
091    int cipherMode = Openssl.DECRYPT_MODE;
092    if (mode == ENCRYPT_MODE)
093      cipherMode = Openssl.ENCRYPT_MODE;
094
095    cipher.init(cipherMode, key, iv);
096  }
097
098  /**
099   * Continues a multiple-part encryption/decryption operation. The data
100   * is encrypted or decrypted, depending on how this cipher was initialized.
101   * @param inBuffer the input ByteBuffer
102   * @param outBuffer the output ByteBuffer
103   * @return int number of bytes stored in <code>output</code>
104   * @throws ShortBufferException if there is insufficient space
105   * in the output buffer
106   */
107  @Override
108  public int update(ByteBuffer inBuffer, ByteBuffer outBuffer)
109      throws ShortBufferException {
110    return cipher.update(inBuffer, outBuffer);
111  }
112
113  /**
114   * Continues a multiple-part encryption/decryption operation. The data
115   * is encrypted or decrypted, depending on how this cipher was initialized.
116   *
117   * @param input the input byte array
118   * @param inputOffset the offset in input where the input starts
119   * @param inputLen the input length
120   * @param output the byte array for the result
121   * @param outputOffset the offset in output where the result is stored
122   * @return the number of bytes stored in output
123   * @throws ShortBufferException if there is insufficient space in the output byte array
124   */
125  @Override
126  public int update(byte[] input, int inputOffset, int inputLen,
127      byte[] output, int outputOffset)
128      throws ShortBufferException {
129    return cipher.update(input, inputOffset, inputLen,
130                         output, outputOffset);
131  }
132
133  /**
134   * Encrypts or decrypts data in a single-part operation, or finishes a
135   * multiple-part operation. The data is encrypted or decrypted, depending
136   * on how this cipher was initialized.
137   * @param inBuffer the input ByteBuffer
138   * @param outBuffer the output ByteBuffer
139   * @return int number of bytes stored in <code>output</code>
140   * @throws BadPaddingException if this cipher is in decryption mode,
141   * and (un)padding has been requested, but the decrypted data is not
142   * bounded by the appropriate padding bytes
143   * @throws IllegalBlockSizeException if this cipher is a block cipher,
144   * no padding has been requested (only in encryption mode), and the total
145   * input length of the data processed by this cipher is not a multiple of
146   * block size; or if this encryption algorithm is unable to
147   * process the input data provided.
148   * @throws ShortBufferException if the given output buffer is too small
149   * to hold the result
150   */
151  @Override
152  public int doFinal(ByteBuffer inBuffer, ByteBuffer outBuffer)
153      throws ShortBufferException, IllegalBlockSizeException,
154      BadPaddingException {
155    int n = cipher.update(inBuffer, outBuffer);
156    return n + cipher.doFinal(outBuffer);
157  }
158
159  /**
160   * Encrypts or decrypts data in a single-part operation, or finishes a
161   * multiple-part operation.
162   *
163   * @param input the input byte array
164   * @param inputOffset the offset in input where the input starts
165   * @param inputLen the input length
166   * @param output the byte array for the result
167   * @param outputOffset the offset in output where the result is stored
168   * @return the number of bytes stored in output
169   * @throws ShortBufferException if the given output byte array is too small
170   * to hold the result
171   * @throws BadPaddingException if this cipher is in decryption mode,
172   * and (un)padding has been requested, but the decrypted data is not
173   * bounded by the appropriate padding bytes
174   * @throws IllegalBlockSizeException if this cipher is a block cipher,
175   * no padding has been requested (only in encryption mode), and the total
176   * input length of the data processed by this cipher is not a multiple of
177   * block size; or if this encryption algorithm is unable to
178   * process the input data provided.
179   */
180  @Override
181  public int doFinal(byte[] input, int inputOffset, int inputLen,
182      byte[] output, int outputOffset)
183      throws ShortBufferException, IllegalBlockSizeException, BadPaddingException {
184    int n = cipher.update(input, inputOffset, inputLen,
185                          output, outputOffset);
186    return n + cipher.doFinal(output, outputOffset + n);
187  }
188
189  /**
190   * Closes the OpenSSL cipher. Clean the Openssl native context.
191   */
192  @Override
193  public void close() {
194    cipher.clean();
195  }
196}
