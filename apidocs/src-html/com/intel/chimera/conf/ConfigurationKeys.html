018package com.intel.chimera.conf;
019
020import com.intel.chimera.cipher.JceCipher;
021
022/**
023 * The ConfigurationKeys contains Configuration keys and default values.
024 */
025public class ConfigurationKeys {
026
027  /**
028   * The prefix named with project name.
029   */
030  public static final String CHIMERA_PREFIX = "chimera.";
031
032  /**
033   * The filename of configuration file.
034   */
035  public static final String CHIMERA_SYSTEM_PROPERTIES_FILE =
036      CHIMERA_PREFIX + "properties";
037
038  /**
039   * The prefix of crypto configuration.
040   */
041  public static final String CONF_PREFIX =
042      CHIMERA_PREFIX + "crypto.";
043
044  /**
045   * The configuration key of implementation class for crypto cipher.
046   * The values of CHIMERA_CRYPTO_CIPHER_CLASSES_KEY can be
047   * "com.intel.chimera.cipher.JceCipher" and "com.intel.chimera.cipher.OpensslCipher".
048   * And it takes a common separated list.
049   * The "com.intel.chimera.cipher.JceCipher" use jce provider to
050   * implement {@link com.intel.chimera.cipher.Cipher} and
051   * the "com.intel.chimera.cipher.OpensslCipher" use jni into openssl to implement.
052   * Note that for each value,the first value which can be created without exception
053   * will be used (priority by order).
054    */
055  public static final String CHIMERA_CRYPTO_CIPHER_CLASSES_KEY =
056      CONF_PREFIX + "cipher.classes";
057
058  /**
059   * The default value for crypto cipher.
060   */
061  public static final String CHIMERA_CRYPTO_CIPHER_CLASSES_DEFAULT =
062      JceCipher.class.getName();
063
064  /**
065   * The configuration key of the provider class for JCE cipher.
066   */
067  public static final String CHIMERA_CRYPTO_CIPHER_JCE_PROVIDER_KEY =
068      CONF_PREFIX + "cipher.jce.provider";
069
070  // security random related configuration keys
071  /**
072   * The configuration key of the file path for secure random device.
073   */
074  public static final String CHIMERA_CRYPTO_SECURE_RANDOM_DEVICE_FILE_PATH_KEY =
075      CONF_PREFIX + "secure.random.device.file.path";
076
077  /**
078   * The default value of the file path for secure random device.
079   */
080  public static final String CHIMERA_CRYPTO_SECURE_RANDOM_DEVICE_FILE_PATH_DEFAULT =
081      "/dev/urandom";
082
083  /**
084   * The configuration key of the algorithm of secure random.
085   */
086  public static final String CHIMERA_CRYPTO_SECURE_RANDOM_JAVA_ALGORITHM_KEY =
087      CONF_PREFIX + "secure.random.java.algorithm";
088
089  /**
090   * The default value of the algorithm of secure random.
091   */
092  public static final String
093      CHIMERA_CRYPTO_SECURE_RANDOM_JAVA_ALGORITHM_DEFAULT = "SHA1PRNG";
094
095  /**
096   * The configuration key of the implementation class for secure random.
097   * The values of CHIMERA_CRYPTO_SECURE_RANDOM_CLASSES_KEY can be
098   * "com.intel.chimera.random.JavaSecureRandom" and "com.intel.chimera.random.OpensslSecureRandom".
099   * And it takes a common separated list.
100   * The "com.intel.chimera.random.JavaSecureRandom" use java to
101   * implement {@link com.intel.chimera.random.SecureRandom} and
102   * the "com.intel.chimera.random.OpensslSecureRandom" use jni into openssl to implement.
103   * Note that for each value,the first value which can be created without exception
104   * will be used (priority by order).
105   */
106  public static final String CHIMERA_CRYPTO_SECURE_RANDOM_CLASSES_KEY =
107      CONF_PREFIX + "secure.random.classes";
108
109  /**
110   * The configuration key of the buffer size for stream.
111   */
112  public static final String CHIMERA_CRYPTO_STREAM_BUFFER_SIZE_KEY =
113      CONF_PREFIX + "stream.buffer.size";
114
115  // stream related configuration keys
116  /**
117   * The default value of the buffer size for stream.
118   */
119  public static final int CHIMERA_CRYPTO_STREAM_BUFFER_SIZE_DEFAULT = 8192;
120
121  // native lib related configuration keys
122  /**
123   * The configuration key of the path for loading crypto library.
124   */
125  public static final String CHIMERA_CRYPTO_LIB_PATH_KEY =
126      CONF_PREFIX + "lib.path";
127
128  /**
129   * The configuration key of the file name for loading crypto library.
130   */
131  public static final String CHIMERA_CRYPTO_LIB_NAME_KEY =
132      CONF_PREFIX + "lib.name";
133
134  /**
135   * The configuration key of temp directory for extracting crypto library.
136   */
137  public static final String CHIMERA_CRYPTO_LIB_TEMPDIR_KEY =
138      CONF_PREFIX + "lib.tempdir";
139}
