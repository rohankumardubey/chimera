018package com.intel.chimera.random;
019
020import java.util.Properties;
021import java.util.Random;
022
023import org.apache.commons.logging.Log;
024import org.apache.commons.logging.LogFactory;
025
026import com.intel.chimera.utils.NativeCodeLoader;
027import com.intel.chimera.utils.Utils;
028
029/**
030 * OpenSSL secure random using JNI.
031 * This implementation is thread-safe.
032 * <p/>
033 *
034 * If using an Intel chipset with RDRAND, the high-performance hardware
035 * random number generator will be used and it's much faster than
036 * {@link java.security.SecureRandom}. If RDRAND is unavailable, default
037 * OpenSSL secure random generator will be used. It's still faster
038 * and can generate strong random bytes.
039 * <p/>
040 * @see https://wiki.openssl.org/index.php/Random_Numbers
041 * @see http://en.wikipedia.org/wiki/RdRand
042 */
043public class OpensslSecureRandom extends Random implements SecureRandom {
044  private static final long serialVersionUID = -7828193502768789584L;
045  private static final Log LOG =
046      LogFactory.getLog(OpensslSecureRandom.class.getName());
047
048  /** If native SecureRandom unavailable, use java SecureRandom */
049  private JavaSecureRandom fallback = null;
050  private static boolean nativeEnabled = false;
051  static {
052    if (NativeCodeLoader.isNativeCodeLoaded()) {
053      try {
054        OpensslSecureRandomNative.initSR();
055        nativeEnabled = true;
056      } catch (Throwable t) {
057        LOG.error("Failed to load Openssl SecureRandom", t);
058      }
059    }
060  }
061
062  /**
063   * Judges whether loading native library successfully.
064   *
065   * @return true if loading library successfully.
066   */
067  public static boolean isNativeCodeLoaded() {
068    return nativeEnabled;
069  }
070
071  /**
072   * Constructs a {@link com.intel.chimera.random.OpensslSecureRandom}.
073   *
074   * @param props the configuration properties.
075   */
076  public OpensslSecureRandom(Properties props) {
077    if (!nativeEnabled) {
078      fallback = new JavaSecureRandom(props);
079    }
080  }
081
082  /**
083   * Generates a user-specified number of random bytes.
084   * It's thread-safe.
085   *
086   * @param bytes the array to be filled in with random bytes.
087   */
088  @Override
089  public void nextBytes(byte[] bytes) {
090    if (!nativeEnabled || !OpensslSecureRandomNative.nextRandBytes(bytes)) {
091      fallback.nextBytes(bytes);
092    }
093  }
094
095  /**
096   * Overrides {@link OpensslSecureRandom}.
097   * For {@link OpensslSecureRandom}, we don't need to set seed.
098   *
099   * @param seed the initial seed.
100   */
101  @Override
102  public void setSeed(long seed) {
103    // Self-seeding.
104  }
105
106  /**
107   * Overrides {@link java.util.Random# next()}. Generates an integer
108   * containing the user-specified number of random
109   * bits(right justified, with leading zeros).
110   *
111   * @param numBits number of random bits to be generated, where
112   * 0 <= <code>numBits</code> <= 32.
113   * @return int an <code>int</code> containing the user-specified number
114   * of random bits (right justified, with leading zeros).
115   */
116  @Override
117  final protected int next(int numBits) {
118    Utils.checkArgument(numBits >= 0 && numBits <= 32);
119    int numBytes = (numBits + 7) / 8;
120    byte b[] = new byte[numBytes];
121    int next = 0;
122
123    nextBytes(b);
124    for (int i = 0; i < numBytes; i++) {
125      next = (next << 8) + (b[i] & 0xFF);
126    }
127
128    return next >>> (numBytes * 8 - numBits);
129  }
130
131  /**
132   * Overrides {@link java.lang.AutoCloseable#close()}. Closes openssl context if native enabled.
133   */
134  @Override
135  public void close() {
136    if (!nativeEnabled && fallback !=null) {
137      fallback.close();
138    }
139  }
140}
