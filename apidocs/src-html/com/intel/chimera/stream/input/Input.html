018package com.intel.chimera.stream.input;
019
020import java.io.IOException;
021import java.nio.ByteBuffer;
022
023/**
024 * The Input interface abstract the input source of <code>CryptoInputStream</code> so that
025 * different implementation of input can be used. The implementation Input interface will usually
026 * wraps an input mechanism such as <code>InputStream</code> or <code>ReadableByteChannel</code>.
027 */
028public interface Input {
029  /**
030   * Reads a sequence of bytes from input into the given buffer.
031   *
032   * <p> An attempt is made to read up to <i>r</i> bytes from the input,
033   * where <i>r</i> is the number of bytes remaining in the buffer, that is,
034   * <tt>dst.remaining()</tt>, at the moment this method is invoked.
035   *
036   * <p> Suppose that a byte sequence of length <i>n</i> is read, where
037   * <tt>0</tt>&nbsp;<tt>&lt;=</tt>&nbsp;<i>n</i>&nbsp;<tt>&lt;=</tt>&nbsp;<i>r</i>.
038   * This byte sequence will be transferred into the buffer so that the first
039   * byte in the sequence is at index <i>p</i> and the last byte is at index
040   * <i>p</i>&nbsp;<tt>+</tt>&nbsp;<i>n</i>&nbsp;<tt>-</tt>&nbsp;<tt>1</tt>,
041   * where <i>p</i> is the buffer's position at the moment this method is
042   * invoked.  Upon return the buffer's position will be equal to
043   * <i>p</i>&nbsp;<tt>+</tt>&nbsp;<i>n</i>; its limit will not have changed.
044   *
045   * @param  dst
046   *         The buffer into which bytes are to be transferred
047   *
048   * @throws  IOException
049   *          If some other I/O error occurs
050   */
051  int read(ByteBuffer dst) throws IOException;
052
053  /**
054   * Skips over and discards <code>n</code> bytes of data from this input
055   * The <code>skip</code> method may, for a variety of reasons, end
056   * up skipping over some smaller number of bytes, possibly <code>0</code>.
057   * This may result from any of a number of conditions; reaching end of file
058   * before <code>n</code> bytes have been skipped is only one possibility.
059   * The actual number of bytes skipped is returned.  If <code>n</code> is
060   * negative, no bytes are skipped.
061   *
062   * <p> The <code>skip</code> method of this class creates a
063   * byte array and then repeatedly reads into it until <code>n</code> bytes
064   * have been read or the end of the stream has been reached. Subclasses are
065   * encouraged to provide a more efficient implementation of this method.
066   * For instance, the implementation may depend on the ability to seek.
067   *
068   * @param      n   the number of bytes to be skipped.
069   * @return     the actual number of bytes skipped.
070   * @exception  IOException  if the stream does not support seek,
071   *                          or if some other I/O error occurs.
072   */
073  long skip(long n) throws IOException;
074
075  /**
076   * Returns an estimate of the number of bytes that can be read (or
077   * skipped over) from this input without blocking by the next
078   * invocation of a method for this input stream. The next invocation
079   * might be the same thread or another thread.  A single read or skip of this
080   * many bytes will not block, but may read or skip fewer bytes.
081   *
082   * <p> It is never correct to use the return value of this method to allocate
083   * a buffer intended to hold all data in this stream.
084   *
085   * @return     an estimate of the number of bytes that can be read (or skipped
086   *             over) from this input stream without blocking or {@code 0} when
087   *             it reaches the end of the input stream.
088   * @exception  IOException if an I/O error occurs.
089   */
090  int available() throws IOException;
091
092  /**
093   * Read up to the specified number of bytes from a given position within a
094   * stream and return the number of bytes read.
095   * This does not change the current offset of the stream and is thread-safe.
096   * 
097   * An implementation may not support positioned read. If the implementation
098   * doesn't support positioned read, it throws UnsupportedOperationException.
099   */
100  int read(long position, byte[] buffer, int offset, int length)
101      throws IOException;
102
103  /**
104   * Seek to the given offset from the start of the stream.
105   * The next read() will be from that location.
106   * 
107   * An implementation may not support seek. If the implementation 
108   * doesn't support seek, it throws UnsupportedOperationException.
109   */
110  void seek(long position) throws IOException;
111
112  /**
113   * Closes this input and releases any system resources associated
114   * with the under layer input.
115   *
116   * @exception  IOException  if an I/O error occurs.
117   */
118  void close() throws IOException;
119}
