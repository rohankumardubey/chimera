018package com.intel.chimera.stream.output;
019
020import java.io.IOException;
021import java.nio.ByteBuffer;
022
023/**
024 * The Output interface abstract the output target of <code>CryptoOutputStream</code> so that
025 * different implementation of output can be used. The implementation Output interface will usually
026 * wraps an output mechanism such as <code>OutputStream</code> or <code>WritableByteChannel</code>.
027 */
028
029public interface Output {
030
031  /**
032   * Writes a sequence of bytes to this output from the given buffer.
033   *
034   * <p> An attempt is made to write up to <i>r</i> bytes to the channel,
035   * where <i>r</i> is the number of bytes remaining in the buffer, that is,
036   * <tt>src.remaining()</tt>, at the moment this method is invoked.
037   *
038   * <p> Suppose that a byte sequence of length <i>n</i> is written, where
039   * <tt>0</tt>&nbsp;<tt>&lt;=</tt>&nbsp;<i>n</i>&nbsp;<tt>&lt;=</tt>&nbsp;<i>r</i>.
040   * This byte sequence will be transferred from the buffer starting at index
041   * <i>p</i>, where <i>p</i> is the buffer's position at the moment this
042   * method is invoked; the index of the last byte written will be
043   * <i>p</i>&nbsp;<tt>+</tt>&nbsp;<i>n</i>&nbsp;<tt>-</tt>&nbsp;<tt>1</tt>.
044   * Upon return the buffer's position will be equal to
045   * <i>p</i>&nbsp;<tt>+</tt>&nbsp;<i>n</i>; its limit will not have changed.
046   *
047   * @param  src
048   *         The buffer from which bytes are to be retrieved
049   *
050   * @return The number of bytes written, possibly zero
051   *
052   * @throws  IOException
053   *          If some other I/O error occurs
054   */
055  int write(ByteBuffer src) throws IOException;
056  
057  /**
058   * Flushes this output and forces any buffered output bytes
059   * to be written out if the under layer output method support
060   * The general contract of <code>flush</code> is
061   * that calling it is an indication that, if any bytes previously
062   * written have been buffered by the implementation of the output
063   * stream, such bytes should immediately be written to their
064   * intended destination.
065   *
066   * @exception  IOException  if an I/O error occurs.
067   */
068  void flush() throws IOException;
069  
070  /**
071   * Closes this output and releases any system resources associated
072   * with the under layer output.
073   *
074   * @exception  IOException  if an I/O error occurs.
075   */
076  void close() throws IOException;
077}
