018package com.intel.chimera.utils;
019
020import java.io.IOException;
021import java.io.InputStream;
022
023import com.intel.chimera.stream.input.Input;
024import org.apache.commons.logging.Log;
025
026public class IOUtils {
027  public static void readFully(InputStream in, byte buf[],
028      int off, int len) throws IOException {
029    int toRead = len;
030    while (toRead > 0) {
031      int ret = in.read(buf, off, toRead);
032      if (ret < 0) {
033        throw new IOException( "Premature EOF from inputStream");
034      }
035      toRead -= ret;
036      off += ret;
037    }
038  }
039
040  /**
041   * Do the readFully based on Input's positioned read.
042   * This does not change the current offset of the stream and is thread-safe.
043   */
044  public static void readFully(Input in, long position,
045      byte[] buffer, int offset, int length) throws IOException {
046    int nread = 0;
047    while (nread < length) {
048      int nbytes = in.read(position+nread, buffer, offset+nread, length-nread);
049      if (nbytes < 0) {
050        throw new IOException("End of stream reached before reading fully.");
051      }
052      nread += nbytes;
053    }
054  }
055
056  /**
057   * Close the Closeable objects and <b>ignore</b> any {@link IOException} or 
058   * null pointers. Must only be used for cleanup in exception handlers.
059   *
060   * @param log the log to record problems to at debug level. Can be null.
061   * @param closeables the objects to close
062   */
063  public static void cleanup(Log log, java.io.Closeable... closeables) {
064    for (java.io.Closeable c : closeables) {
065      if (c != null) {
066        try {
067          c.close();
068        } catch(Throwable e) {
069          if (log != null && log.isDebugEnabled()) {
070            log.debug("Exception in closing " + c, e);
071          }
072        }
073      }
074    }
075  }
076}
