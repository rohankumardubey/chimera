018package com.intel.chimera.utils;
019
020import java.lang.ref.WeakReference;
021import java.lang.reflect.Constructor;
022import java.util.Collections;
023import java.util.Map;
024import java.util.WeakHashMap;
025
026import com.intel.chimera.cipher.Cipher;
027
028public class ReflectionUtils {
029
030  private static final Map<ClassLoader, Map<String, WeakReference<Class<?>>>>
031    CACHE_CLASSES = new WeakHashMap<ClassLoader, Map<String, WeakReference<Class<?>>>>();
032
033  private static ClassLoader classLoader;
034  static {
035    classLoader = Thread.currentThread().getContextClassLoader();
036    if (classLoader == null) {
037      classLoader = Cipher.class.getClassLoader();
038    }
039  }
040
041  /**
042   * Sentinel value to store negative cache results in {@link #CACHE_CLASSES}.
043   */
044  private static final Class<?> NEGATIVE_CACHE_SENTINEL =
045    NegativeCacheSentinel.class;
046
047  /**
048   * A unique class which is used as a sentinel value in the caching
049   * for getClassByName. {@link Cipher#getClassByNameOrNull(String)}
050   */
051  private static abstract class NegativeCacheSentinel {}
052
053  @SuppressWarnings("rawtypes")
054  public static <T> T newInstance(Class<T> klass, Object ... args) {
055    try {
056      Constructor<T> ctor = null;
057
058      if (args.length == 0) {
059        ctor = klass.getDeclaredConstructor(new Class[] {});
060      } else {
061        Class[] argClses = new Class[args.length];
062        for (int i = 0; i < args.length; i++) {
063          argClses[i] = args[i].getClass();
064        }
065        ctor = klass.getDeclaredConstructor(argClses);
066      }
067      ctor.setAccessible(true);
068      return ctor.newInstance(args);
069    } catch (Exception e) {
070      throw new RuntimeException(e);
071    }
072  }
073
074  /** 
075   * Get the value of the <code>name</code> property as a <code>Class</code>
076   * implementing the interface specified by <code>xface</code>.
077   *   
078   * If no such property is specified, then <code>defaultValue</code> is 
079   * returned.
080   * 
081   * An exception is thrown if the returned class does not implement the named
082   * interface. 
083   * 
084   * @param name the class name.
085   * @param defaultValue default value.
086   * @param xface the interface implemented by the named class.
087   * @return property value as a <code>Class</code>, 
088   *         or <code>defaultValue</code>.
089   */
090  public static <U> Class<? extends U> getClass(String name, 
091                                         Class<? extends U> defaultValue, 
092                                         Class<U> xface) {
093    try {
094      Class<?> theClass = null;
095      if (name != null && !name.isEmpty()) {
096        theClass = getClassByName(name);
097      }
098      if (theClass == null) {
099        theClass = defaultValue;
100      }
101      if (theClass != null && !xface.isAssignableFrom(theClass))
102        throw new RuntimeException(theClass+" not "+xface.getName());
103      else if (theClass != null)
104        return theClass.asSubclass(xface);
105      else
106        return null;
107    } catch (Exception e) {
108      throw new RuntimeException(e);
109    }
110  }
111
112  /** 
113   * Get the value of the <code>name</code> property as a <code>Class</code>.  
114   * If no such property is specified, then <code>defaultValue</code> is 
115   * returned.
116   * 
117   * @param name the class name.
118   * @param defaultValue default value.
119   * @return property value as a <code>Class</code>, 
120   *         or <code>defaultValue</code>. 
121   */
122  public static Class<?> getClass(String name, Class<?> defaultValue) {
123    String valueString = System.getProperty(name);
124    if (valueString == null)
125      return defaultValue;
126    try {
127      return getClassByName(valueString);
128    } catch (ClassNotFoundException e) {
129      throw new RuntimeException(e);
130    }
131  }
132
133  /**
134   * Load a class by name.
135   * 
136   * @param name the class name.
137   * @return the class object.
138   * @throws ClassNotFoundException if the class is not found.
139   */
140  public static Class<?> getClassByName(String name)
141      throws ClassNotFoundException {
142    Class<?> ret = getClassByNameOrNull(name);
143    if (ret == null) {
144      throw new ClassNotFoundException("Class " + name + " not found");
145    }
146    return ret;
147  }
148  
149  /**
150   * Load a class by name, returning null rather than throwing an exception
151   * if it couldn't be loaded. This is to avoid the overhead of creating
152   * an exception.
153   * 
154   * @param name the class name
155   * @return the class object, or null if it could not be found.
156   */
157  private static Class<?> getClassByNameOrNull(String name) {
158    Map<String, WeakReference<Class<?>>> map;
159
160    synchronized (CACHE_CLASSES) {
161      map = CACHE_CLASSES.get(classLoader);
162      if (map == null) {
163        map = Collections.synchronizedMap(
164          new WeakHashMap<String, WeakReference<Class<?>>>());
165        CACHE_CLASSES.put(classLoader, map);
166      }
167    }
168
169    Class<?> clazz = null;
170    WeakReference<Class<?>> ref = map.get(name); 
171    if (ref != null) {
172       clazz = ref.get();
173    }
174     
175    if (clazz == null) {
176      try {
177        clazz = Class.forName(name, true, classLoader);
178      } catch (ClassNotFoundException e) {
179        // Leave a marker that the class isn't found
180        map.put(name, new WeakReference<Class<?>>(NEGATIVE_CACHE_SENTINEL));
181        return null;
182      }
183      // two putters can race here, but they'll put the same class
184      map.put(name, new WeakReference<Class<?>>(clazz));
185      return clazz;
186    } else if (clazz == NEGATIVE_CACHE_SENTINEL) {
187      return null; // not found
188    } else {
189      // cache hit
190      return clazz;
191    }
192  }
193}
