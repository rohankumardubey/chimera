060  /**
061   * load system properties when configuration file of the name
062   * {@link #CHIMERA_SYSTEM_PROPERTIES_FILE} is found
063   */
064  private static void loadChimeraSystemProperties() {
065    try {
066      InputStream is = Thread.currentThread().getContextClassLoader()
067          .getResourceAsStream(CHIMERA_SYSTEM_PROPERTIES_FILE);
068
069      if (is == null)
070        return; // no configuration file is found
071
072      // Load property file
073      Properties props = new Properties();
074      props.load(is);
075      is.close();
076      Enumeration<?> names = props.propertyNames();
077      while (names.hasMoreElements()) {
078        String name = (String) names.nextElement();
079        if (name.startsWith("chimera.")) {
080          if (System.getProperty(name) == null) {
081            System.setProperty(name, props.getProperty(name));
082          }
083        }
084      }
085    } catch (Throwable ex) {
086      System.err.println("Could not load '"
087          + CHIMERA_SYSTEM_PROPERTIES_FILE + "' from classpath: "
088          + ex.toString());
089    }
090  }

092  /** Forcibly free the direct buffer. */
093  public static void freeDirectBuffer(ByteBuffer buffer) {
094    if (buffer instanceof sun.nio.ch.DirectBuffer) {
095      final sun.misc.Cleaner bufferCleaner =
096          ((sun.nio.ch.DirectBuffer) buffer).cleaner();
097      bufferCleaner.clean();
098    }

101  /** Read crypto buffer size */

103    String bufferSizeStr = props.getProperty(
104        CHIMERA_CRYPTO_STREAM_BUFFER_SIZE_KEY);
105    if (bufferSizeStr == null || bufferSizeStr.isEmpty()) {
106      bufferSizeStr = System
107        .getProperty(CHIMERA_CRYPTO_STREAM_BUFFER_SIZE_KEY);
108    }
109    if (bufferSizeStr == null || bufferSizeStr.isEmpty()) {
110      return CHIMERA_CRYPTO_STREAM_BUFFER_SIZE_DEFAULT;
111    } else {
112      return Integer.parseInt(bufferSizeStr);
113    }
114  }

152  /** AES/CTR/NoPadding is required */
153  public static void checkStreamCipher(Cipher cipher) throws IOException {

155      throw new IOException("AES/CTR/NoPadding is required");

159  /** Check and floor buffer size */
160  public static int checkBufferSize(Cipher cipher, int bufferSize) {
161    checkArgument(bufferSize >= MIN_BUFFER_SIZE,
162        "Minimum value of buffer size is " + MIN_BUFFER_SIZE + ".");
163    return bufferSize - bufferSize % cipher.getTransformation()
164        .getAlgorithmBlockSize();
165  }

167  /**
168   * This method is only for Counter (CTR) mode. Generally the Cipher calculates the IV and maintain encryption context internally.
169   * For example a {@link javax.crypto.Cipher} will maintain its encryption
170   * context internally when we do encryption/decryption using the
171   * Cipher#update interface.
172   * <p/>
173   * Encryption/Decryption is not always on the entire file. For example,
174   * in Hadoop, a node may only decrypt a portion of a file (i.e. a split).
175   * In these situations, the counter is derived from the file position.
176   * <p/>
177   * The IV can be calculated by combining the initial IV and the counter with
178   * a lossless operation (concatenation, addition, or XOR).
179   * @see http://en.wikipedia.org/wiki/Block_cipher_mode_of_operation#Counter_.28CTR.29
180   *
181   * @param initIV initial IV
182   * @param counter counter for input stream position
183   * @param IV the IV for input stream position
184   */
185  public static void calculateIV(byte[] initIV, long counter, byte[] IV) {
186    checkArgument(initIV.length == AES_BLOCK_SIZE);
187    checkArgument(IV.length == AES_BLOCK_SIZE);
188
189    int i = IV.length; // IV length
190    int j = 0; // counter bytes index
191    int sum = 0;
192    while (i-- > 0) {
193      // (sum >>> Byte.SIZE) is the carry for addition
194      sum = (initIV[i] & 0xff) + (sum >>> Byte.SIZE);
195      if (j++ < 8) { // Big-endian, and long is 8 bytes length
196        sum += (byte) counter & 0xff;
197        counter >>>= 8;
198      }
199      IV[i] = (byte) sum;
200    }
201  }

206  public static Cipher getCipherInstance(CipherTransformation transformation,

216   * Ensures the truth of an expression involving one or more parameters to
217   * the calling method.
218   *
219   * @param expression a boolean expression
220   * @throws IllegalArgumentException if expression is false
221   */
222  public static void checkArgument(boolean expression) {
223    if(!expression) {
224      throw new IllegalArgumentException();
225    }

228  /**
229   * @param expression   a boolean expression
230   * @param errorMessage the exception message to use if the check fails;
231   *                     will be converted to a string using <code>String
232   *                     .valueOf(Object)</code>
233   * @throws IllegalArgumentException if expression is false
234   */
235  public static void checkArgument(boolean expression, Object errorMessage) {
236    if (!expression) {
237      throw new IllegalArgumentException(String.valueOf(errorMessage));

241  /**
242   * Ensures that an object reference passed as a parameter to the calling
243   * method is not null.
244   *

246   * @return the non-null reference that was validated

257  /**
258   * Ensures the truth of an expression involving the state of the calling
259   * instance, but not involving any parameters to the calling method.

261   * @param expression a boolean expression
262   * @throws IllegalStateException if expression is false

264  public static void checkState(boolean expression) {
265    if(!expression) {
266      throw new IllegalStateException();
267    }

270  /**
271   * Split class names sequence into substrings, Trim each substring into an
272   * entry,and returns an list of the entries.
273   *
274   * @param clazzNames a string consist of a list of the entries joined by a
275   *                   delimiter
276   * @param separator  a delimiter for the input string
277   * @return a list of entries
278   */
279  public static List<String> splitClassNames(String clazzNames,
280      String separator) {
281    List<String> res = new ArrayList<String>();
282    if (clazzNames == null || clazzNames.isEmpty()) {
283      return res;
284    }
285
286    for (String clazzName : clazzNames.split(separator)) {
287      clazzName = clazzName.trim();
288      if (!clazzName.isEmpty()) {
289        res.add(clazzName);
290      }
291    }
292    return res;
293  }
