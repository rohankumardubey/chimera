106   private void doSkipTest(String cipherClass, boolean withChannel) throws IOException {
107     InputStream in = getCryptoInputStream(new ByteArrayInputStream(encData),
108         getCipher(cipherClass), defaultBufferSize, iv, withChannel);
109     byte[] result = new byte[dataLen];
110     int n1 = readAll(in, result, 0, dataLen / 3);
111 
112     long skipped = in.skip(dataLen / 3);
113     int n2 = readAll(in, result, 0, dataLen);
114 
115     Assert.assertEquals(dataLen, n1 + skipped + n2);
116     byte[] readData = new byte[n2];
117     System.arraycopy(result, 0, readData, 0, n2);
118     byte[] expectedData = new byte[n2];
119     System.arraycopy(data, dataLen - n2, expectedData, 0, n2);
120     Assert.assertArrayEquals(readData, expectedData);
121 
122     try {
123       skipped = in.skip(-3);
124       Assert.fail("Skip Negative length should fail.");
125     } catch (IllegalArgumentException e) {
126       Assert.assertTrue(e.getMessage().contains("Negative skip length"));
127     }
128 
129     // Skip after EOF
130     skipped = in.skip(3);
131     Assert.assertEquals(skipped, 0);
132 
133     in.close();
134   }

136   private void doByteBufferRead(String cipherClass, boolean withChannel) throws Exception {
137     // Default buffer size, initial buffer position is 0
138     InputStream in = getCryptoInputStream(new ByteArrayInputStream(encData),
139         getCipher(cipherClass), defaultBufferSize, iv, withChannel);
140     ByteBuffer buf = ByteBuffer.allocate(dataLen + 100);
141     byteBufferReadCheck(in, buf, 0);
142     in.close();
143 
144     // Default buffer size, initial buffer position is not 0
145     in = getCryptoInputStream(new ByteArrayInputStream(encData),
146         getCipher(cipherClass), defaultBufferSize, iv, withChannel);
147     buf.clear();
148     byteBufferReadCheck(in, buf, 11);
149     in.close();
150 
151     // Small buffer size, initial buffer position is 0
152     in = getCryptoInputStream(new ByteArrayInputStream(encData),
153         getCipher(cipherClass), smallBufferSize, iv, withChannel);
154     buf.clear();
155     byteBufferReadCheck(in, buf, 0);
156     in.close();
157 
158     // Small buffer size, initial buffer position is not 0
159     in = getCryptoInputStream(new ByteArrayInputStream(encData),
160         getCipher(cipherClass), smallBufferSize, iv, withChannel);
161     buf.clear();
162     byteBufferReadCheck(in, buf, 11);
163     in.close();
164 
165     // Direct buffer, default buffer size, initial buffer position is 0
166     in = getCryptoInputStream(new ByteArrayInputStream(encData),
167         getCipher(cipherClass), defaultBufferSize, iv, withChannel);
168     buf = ByteBuffer.allocateDirect(dataLen + 100);
169     byteBufferReadCheck(in, buf, 0);
170     in.close();
171 
172     // Direct buffer, default buffer size, initial buffer position is not 0
173     in = getCryptoInputStream(new ByteArrayInputStream(encData),
174         getCipher(cipherClass), defaultBufferSize, iv, withChannel);
175     buf.clear();
176     byteBufferReadCheck(in, buf, 11);
177     in.close();
178 
179     // Direct buffer, small buffer size, initial buffer position is 0
180     in = getCryptoInputStream(new ByteArrayInputStream(encData),
181         getCipher(cipherClass), smallBufferSize, iv, withChannel);
182     buf.clear();
183     byteBufferReadCheck(in, buf, 0);
184     in.close();
185 
186     // Direct buffer, small buffer size, initial buffer position is not 0
187     in = getCryptoInputStream(new ByteArrayInputStream(encData),
188         getCipher(cipherClass), smallBufferSize, iv, withChannel);
189     buf.clear();
190     byteBufferReadCheck(in, buf, 11);
191     in.close();
192   }

194   private void doByteBufferWrite(String cipherClass, ByteArrayOutputStream baos,
195                                  boolean withChannel)
196       throws Exception {
197     baos.reset();
198     CryptoOutputStream out =
199         getCryptoOutputStream(baos, getCipher(cipherClass), defaultBufferSize,
200             iv, withChannel);
201     ByteBuffer buf = ByteBuffer.allocateDirect(dataLen / 2);
202     buf.put(data, 0, dataLen / 2);
203     buf.flip();
204     int n1 = out.write(buf);
205 
206     buf.clear();
207     buf.put(data, n1, dataLen / 3);
208     buf.flip();
209     int n2 = out.write(buf);
210 
211     buf.clear();
212     buf.put(data, n1 + n2, dataLen - n1 - n2);
213     buf.flip();
214     int n3 = out.write(buf);
215 
216     Assert.assertEquals(dataLen, n1 + n2 + n3);
217 
218     out.flush();
219 
220     InputStream in = getCryptoInputStream(new ByteArrayInputStream(encData),
221         getCipher(cipherClass), defaultBufferSize, iv, withChannel);
222     buf = ByteBuffer.allocate(dataLen + 100);
223     byteBufferReadCheck(in, buf, 0);
224     in.close();
225   }

227   private void byteBufferReadCheck(InputStream in, ByteBuffer buf,
228       int bufPos) throws Exception {
229     buf.position(bufPos);
230     int n = ((ReadableByteChannel) in).read(buf);
231     Assert.assertEquals(bufPos + n, buf.position());
232     byte[] readData = new byte[n];
233     buf.rewind();
234     buf.position(bufPos);
235     buf.get(readData);
236     byte[] expectedData = new byte[n];
237     System.arraycopy(data, 0, expectedData, 0, n);
238     Assert.assertArrayEquals(readData, expectedData);
239   }

241   private void prepareData() throws IOException {
242     Cipher cipher = null;
243     try {
244       cipher = (Cipher)ReflectionUtils.newInstance(
245           ReflectionUtils.getClassByName(jceCipherClass), props, transformation);
246     } catch (ClassNotFoundException cnfe) {
247       throw new IOException("Illegal crypto cipher!");
248     }
249 
250     ByteArrayOutputStream baos = new ByteArrayOutputStream();
251     OutputStream out = new CryptoOutputStream(baos, cipher, defaultBufferSize, key, iv);
252     out.write(data);
253     out.flush();
254     out.close();
255     encData = baos.toByteArray();
256   }

258   protected CryptoInputStream getCryptoInputStream(ByteArrayInputStream bais,
259                                                    Cipher cipher,
260                                                    int bufferSize, byte[] iv,
261                                                    boolean withChannel) throws
262       IOException {
263     if (withChannel) {
264       return new CryptoInputStream(Channels.newChannel(bais), cipher,
265           bufferSize, key, iv);
266     } else {
267       return new CryptoInputStream(bais, cipher, bufferSize, key, iv);
268     }
269   }

274                                                    boolean withChannel) throws
275       IOException {
276     if (withChannel) {
277       return new CryptoOutputStream(Channels.newChannel(baos), cipher,
278           bufferSize, key, iv);
279     } else {
280       return new CryptoOutputStream(baos, cipher, bufferSize, key, iv);
281     }

284   private int readAll(InputStream in, byte[] b, int offset, int len)
285       throws IOException {
286     int n = 0;
287     int total = 0;
288     while (n != -1) {
289       total += n;
290       if (total >= len) {
291         break;
292       }
293       n = in.read(b, offset + total, len - total);
294     }
295 
296     return total;
297   }

304       throw new IOException("Illegal crypto cipher!");
305     }

308   @Test
309   public void testReadWrite() throws Exception {
310     Assert.assertEquals(null, Openssl.getLoadingFailureReason());
311     doReadWriteTest(0, jceCipherClass, jceCipherClass, iv);
312     doReadWriteTest(0, opensslCipherClass, opensslCipherClass, iv);
313     doReadWriteTest(count, jceCipherClass, jceCipherClass, iv);
314     doReadWriteTest(count, opensslCipherClass, opensslCipherClass, iv);
315     doReadWriteTest(count, jceCipherClass, opensslCipherClass, iv);
316     doReadWriteTest(count, opensslCipherClass, jceCipherClass, iv);
317     // Overflow test, IV: xx xx xx xx xx xx xx xx ff ff ff ff ff ff ff ff
318     for(int i = 0; i < 8; i++) {
319       iv[8 + i] = (byte) 0xff;
320     }
321     doReadWriteTest(count, jceCipherClass, jceCipherClass, iv);
322     doReadWriteTest(count, opensslCipherClass, opensslCipherClass, iv);
323     doReadWriteTest(count, jceCipherClass, opensslCipherClass, iv);
324     doReadWriteTest(count, opensslCipherClass, jceCipherClass, iv);
325   }

327   private void doReadWriteTest(int count, String encCipherClass,
328                                String decCipherClass, byte[] iv) throws IOException {
329     doReadWriteTestForInputStream(count, encCipherClass, decCipherClass, iv);
330     doReadWriteTestForReadableByteChannel(count, encCipherClass, decCipherClass,
331         iv);
332   }

334   private void doReadWriteTestForInputStream(int count, String encCipherClass,
335                                              String decCipherClass, byte[] iv) throws IOException {
336     Cipher encCipher = getCipher(encCipherClass);
337     LOG.debug("Created a cipher object of type: " + encCipherClass);
338 
339     // Generate data
340     SecureRandom random = new SecureRandom();
341     byte[] originalData = new byte[count];
342     byte[] decryptedData = new byte[count];
343     random.nextBytes(originalData);
344     LOG.debug("Generated " + count + " records");
345 
346     // Encrypt data
347     ByteArrayOutputStream encryptedData = new ByteArrayOutputStream();
348     CryptoOutputStream out =
349         getCryptoOutputStream(encryptedData, encCipher, defaultBufferSize, iv,
350             false);
351     out.write(originalData, 0, originalData.length);
352     out.flush();
353     out.close();
354     LOG.debug("Finished encrypting data");
355 
356     Cipher decCipher = getCipher(decCipherClass);
357     LOG.debug("Created a cipher object of type: " + decCipherClass);
358 
359     // Decrypt data
360     CryptoInputStream in = getCryptoInputStream(
361         new ByteArrayInputStream(encryptedData.toByteArray()), decCipher,
362         defaultBufferSize, iv, false);
363 
364     // Check
365     int remainingToRead = count;
366     int offset = 0;
367     while (remainingToRead > 0) {
368       int n = in.read(decryptedData, offset, decryptedData.length - offset);
369       if (n >=0) {
370         remainingToRead -= n;
371         offset += n;
372       }
373     }
374 
375     Assert.assertArrayEquals("originalData and decryptedData not equal",
376         originalData, decryptedData);
377 
378     // Decrypt data byte-at-a-time
379     in = getCryptoInputStream(
380         new ByteArrayInputStream(encryptedData.toByteArray()), decCipher,
381         defaultBufferSize, iv, false);
382 
383     // Check
384     DataInputStream originalIn = new DataInputStream(new BufferedInputStream(new ByteArrayInputStream(originalData)));
385     int expected;
386     do {
387       expected = originalIn.read();
388       Assert.assertEquals("Decrypted stream read by byte does not match",
389           expected, in.read());
390     } while (expected != -1);
391 
392     LOG.debug("SUCCESS! Completed checking " + count + " records");
393   }
