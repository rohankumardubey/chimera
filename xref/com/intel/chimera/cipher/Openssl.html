35  /**
36   * OpenSSL cryptographic wrapper using JNI.
37   * Currently only AES-CTR is supported. It's flexible to add
38   * other crypto algorithms/modes.
39   */
40  public final class Openssl {
41    private static final Log LOG = LogFactory.getLog(Openssl.class.getName());
42  
43    // Mode constant defined by Openssl JNI
44    public static final int ENCRYPT_MODE = 1;
45    public static final int DECRYPT_MODE = 0;
46  
47    /** Currently only support AES/CTR/NoPadding. */
48    private static enum AlgorithmMode {
49      AES_CTR,
50      AES_CBC;
51  
52      static int get(String algorithm, String mode)
53          throws NoSuchAlgorithmException {
54        try {
55          return AlgorithmMode.valueOf(algorithm + "_" + mode).ordinal();
56        } catch (Exception e) {
57          throw new NoSuchAlgorithmException("Doesn't support algorithm: " +
58              algorithm + " and mode: " + mode);
59        }
60      }
61    }
62  
63    private static enum Padding {
64      NoPadding,
65      PKCS5Padding;
66  
67      static int get(String padding) throws NoSuchPaddingException {
68        try {
69          return Padding.valueOf(padding).ordinal();
70        } catch (Exception e) {
71          throw new NoSuchPaddingException("Doesn't support padding: " + padding);
72        }
73      }
74    }
75  
76    private long context = 0;
77    private final int algorithm;
78    private final int padding;
79  
80    private static final String loadingFailureReason;
81  
82    static {
83      String loadingFailure = null;
84      try {
85        if (NativeCodeLoader.isNativeCodeLoaded()) {
86          OpensslNative.initIDs();
87        }
88      } catch (Throwable t) {
89        loadingFailure = t.getMessage();
90        LOG.debug("Failed to load OpenSSL Cipher.", t);
91      } finally {
92        loadingFailureReason = loadingFailure;
93      }
94    }
95  
96    /**
97     * Gets the failure reason when loading Openssl native.
98     * @return the failure reason.
99     */
100   public static String getLoadingFailureReason() {
101     return loadingFailureReason;
102   }

104   private Openssl(long context, int algorithm, int padding) {
105     this.context = context;
106     this.algorithm = algorithm;
107     this.padding = padding;
108   }

110   /**
111    * Return an <code>OpensslCipher<code> object that implements the specified
112    * transformation.
113    *
114    * @param transformation the name of the transformation, e.g.,
115    * AES/CTR/NoPadding.
116    * @return OpensslCipher an <code>OpensslCipher<code> object
117    * @throws NoSuchAlgorithmException if <code>transformation</code> is null,
118    * empty, in an invalid format, or if Openssl doesn't implement the
119    * specified algorithm.
120    * @throws NoSuchPaddingException if <code>transformation</code> contains
121    * a padding scheme that is not available.
122    */
123   public static final Openssl getInstance(String transformation)
124       throws NoSuchAlgorithmException, NoSuchPaddingException {
125     Transform transform = tokenizeTransformation(transformation);
126     int algorithmMode = AlgorithmMode.get(transform.algorithm, transform.mode);
127     int padding = Padding.get(transform.padding);
128     long context = OpensslNative.initContext(algorithmMode, padding);
129     return new Openssl(context, algorithmMode, padding);
130   }
131 
132   /** Nested class for algorithm, mode and padding. */
133   private static class Transform {
134     final String algorithm;
135     final String mode;
136     final String padding;
137 
138     public Transform(String algorithm, String mode, String padding) {
139       this.algorithm = algorithm;
140       this.mode = mode;
141       this.padding = padding;
142     }
143   }
144 
145   private static Transform tokenizeTransformation(String transformation)
146       throws NoSuchAlgorithmException {
147     if (transformation == null) {
148       throw new NoSuchAlgorithmException("No transformation given.");
149     }
150 
151     /*
152      * Array containing the components of a Cipher transformation:
153      *
154      * index 0: algorithm (e.g., AES)
155      * index 1: mode (e.g., CTR)
156      * index 2: padding (e.g., NoPadding)
157      */
158     String[] parts = new String[3];
159     int count = 0;
160     StringTokenizer parser = new StringTokenizer(transformation, "/");
161     while (parser.hasMoreTokens() && count < 3) {
162       parts[count++] = parser.nextToken().trim();
163     }
164     if (count != 3 || parser.hasMoreTokens()) {
165       throw new NoSuchAlgorithmException("Invalid transformation format: " +
166           transformation);
167     }
168     return new Transform(parts[0], parts[1], parts[2]);
169   }

171   /**
172    * Initialize this cipher with a key and IV.
173    *
174    * @param mode {@link #ENCRYPT_MODE} or {@link #DECRYPT_MODE}
175    * @param key crypto key
176    * @param iv crypto iv
177    */
178   public void init(int mode, byte[] key, byte[] iv) {
179     context = OpensslNative.init(context, mode, algorithm, padding, key, iv);
180   }
181 
182   /**
183    * Continues a multiple-part encryption or decryption operation. The data
184    * is encrypted or decrypted, depending on how this cipher was initialized.
185    * <p/>
186    *
187    * All <code>input.remaining()</code> bytes starting at
188    * <code>input.position()</code> are processed. The result is stored in
189    * the output buffer.
190    * <p/>
191    *
192    * Upon return, the input buffer's position will be equal to its limit;
193    * its limit will not have changed. The output buffer's position will have
194    * advanced by n, when n is the value returned by this method; the output
195    * buffer's limit will not have changed.
196    * <p/>
197    *
198    * If <code>output.remaining()</code> bytes are insufficient to hold the
199    * result, a <code>ShortBufferException</code> is thrown.
200    *
201    * @param input the input ByteBuffer
202    * @param output the output ByteBuffer
203    * @return int number of bytes stored in <code>output</code>
204    * @throws ShortBufferException if there is insufficient space in the
205    * output buffer
206    */
207   public int update(ByteBuffer input, ByteBuffer output)
208       throws ShortBufferException {
209     checkState();
210     Utils.checkArgument(input.isDirect() && output.isDirect(),
211         "Direct buffers are required.");
212     int len = OpensslNative.update(context, input, input.position(),
213         input.remaining(), output, output.position(), output.remaining());
214     input.position(input.limit());
215     output.position(output.position() + len);
216     return len;
217   }
218 
219   /**
220    * Continues a multiple-part encryption/decryption operation. The data
221    * is encrypted or decrypted, depending on how this cipher was initialized.
222    *
223    * @param input the input byte array
224    * @param inputOffset the offset in input where the input starts
225    * @param inputLen the input length
226    * @param output the byte array for the result
227    * @param outputOffset the offset in output where the result is stored
228    * @return the number of bytes stored in output
229    * @throws ShortBufferException if there is insufficient space in the output byte array
230    */
231   public int update(byte[] input, int inputOffset, int inputLen,
232       byte[] output, int outputOffset)
233       throws ShortBufferException {
234     checkState();
235     return OpensslNative.updateByteArray(context, input, inputOffset, inputLen,
236         output, outputOffset, output.length - outputOffset);
237   }
238 
239   /**
240    * Finishes a multiple-part operation. The data is encrypted or decrypted,
241    * depending on how this cipher was initialized.
242    * <p/>
243    *
244    * The result is stored in the output buffer. Upon return, the output buffer's
245    * position will have advanced by n, where n is the value returned by this
246    * method; the output buffer's limit will not have changed.
247    * <p/>
248    *
249    * If <code>output.remaining()</code> bytes are insufficient to hold the result,
250    * a <code>ShortBufferException</code> is thrown.
251    * <p/>
252    *
253    * Upon finishing, this method resets this cipher object to the state it was
254    * in when previously initialized. That is, the object is available to encrypt
255    * or decrypt more data.
256    * <p/>
257    *
258    * If any exception is thrown, this cipher object need to be reset before it
259    * can be used again.
260    *
261    * @param output the output ByteBuffer
262    * @return int number of bytes stored in <code>output</code>
263    * @throws ShortBufferException
264    * @throws IllegalBlockSizeException
265    * @throws BadPaddingException
266    */
267   public int doFinal(ByteBuffer output)
268       throws ShortBufferException, IllegalBlockSizeException,
269       BadPaddingException {
270     checkState();
271     Utils.checkArgument(output.isDirect(), "Direct buffer is required.");
272     int len = OpensslNative.doFinal(context, output, output.position(), output.remaining());
273     output.position(output.position() + len);
274     return len;
275   }
276 
277   /**
278    * Encrypts or decrypts data in a single-part operation, or finishes a
279    * multiple-part operation.
280    *
281    * @param output the byte array for the result
282    * @param outputOffset the offset in output where the result is stored
283    * @return the number of bytes stored in output
284    * @throws ShortBufferException if the given output byte array is too small
285    * to hold the result
286    * @throws BadPaddingException if this cipher is in decryption mode,
287    * and (un)padding has been requested, but the decrypted data is not
288    * bounded by the appropriate padding bytes
289    * @throws IllegalBlockSizeException if this cipher is a block cipher,
290    * no padding has been requested (only in encryption mode), and the total
291    * input length of the data processed by this cipher is not a multiple of
292    * block size; or if this encryption algorithm is unable to
293    * process the input data provided.
294    */
295   public int doFinal(byte[] output, int outputOffset)
296       throws ShortBufferException, IllegalBlockSizeException, BadPaddingException {
297     checkState();
298     return OpensslNative.doFinalByteArray(context,
299         output, outputOffset, output.length - outputOffset);
300   }
301 
302   /** Forcibly clean the context. */
303   public void clean() {
304     if (context != 0) {
305       OpensslNative.clean(context);
306       context = 0;
307     }
308   }
309 
310   /** Checks whether context is initialized. */
311   private void checkState() {
312     Utils.checkState(context != 0);
313   }
314 
315   @Override
316   protected void finalize() throws Throwable {
317     clean();
318   }
