18  package com.intel.chimera.stream.input;
19  
20  import java.io.IOException;
21  import java.nio.ByteBuffer;
22  
23  /**
24   * The Input interface abstract the input source of <code>CryptoInputStream</code> so that
25   * different implementation of input can be used. The implementation Input interface will usually
26   * wraps an input mechanism such as <code>InputStream</code> or <code>ReadableByteChannel</code>.
27   */
28  public interface Input {
29    /**
30     * Reads a sequence of bytes from input into the given buffer.
31     *
32     * <p> An attempt is made to read up to <i>r</i> bytes from the input,
33     * where <i>r</i> is the number of bytes remaining in the buffer, that is,
34     * <tt>dst.remaining()</tt>, at the moment this method is invoked.
35     *
36     * <p> Suppose that a byte sequence of length <i>n</i> is read, where
37     * <tt>0</tt>&nbsp;<tt>&lt;=</tt>&nbsp;<i>n</i>&nbsp;<tt>&lt;=</tt>&nbsp;<i>r</i>.
38     * This byte sequence will be transferred into the buffer so that the first
39     * byte in the sequence is at index <i>p</i> and the last byte is at index
40     * <i>p</i>&nbsp;<tt>+</tt>&nbsp;<i>n</i>&nbsp;<tt>-</tt>&nbsp;<tt>1</tt>,
41     * where <i>p</i> is the buffer's position at the moment this method is
42     * invoked.  Upon return the buffer's position will be equal to
43     * <i>p</i>&nbsp;<tt>+</tt>&nbsp;<i>n</i>; its limit will not have changed.
44     *
45     * @param  dst
46     *         The buffer into which bytes are to be transferred
47     *
48     * @throws  IOException
49     *          If some other I/O error occurs
50     */
51    int read(ByteBuffer dst) throws IOException;
52  
53    /**
54     * Skips over and discards <code>n</code> bytes of data from this input
55     * The <code>skip</code> method may, for a variety of reasons, end
56     * up skipping over some smaller number of bytes, possibly <code>0</code>.
57     * This may result from any of a number of conditions; reaching end of file
58     * before <code>n</code> bytes have been skipped is only one possibility.
59     * The actual number of bytes skipped is returned.  If <code>n</code> is
60     * negative, no bytes are skipped.
61     *
62     * <p> The <code>skip</code> method of this class creates a
63     * byte array and then repeatedly reads into it until <code>n</code> bytes
64     * have been read or the end of the stream has been reached. Subclasses are
65     * encouraged to provide a more efficient implementation of this method.
66     * For instance, the implementation may depend on the ability to seek.
67     *
68     * @param      n   the number of bytes to be skipped.
69     * @return     the actual number of bytes skipped.
70     * @exception  IOException  if the stream does not support seek,
71     *                          or if some other I/O error occurs.
72     */
73    long skip(long n) throws IOException;
74  
75    /**
76     * Returns an estimate of the number of bytes that can be read (or
77     * skipped over) from this input without blocking by the next
78     * invocation of a method for this input stream. The next invocation
79     * might be the same thread or another thread.  A single read or skip of this
80     * many bytes will not block, but may read or skip fewer bytes.
81     *
82     * <p> It is never correct to use the return value of this method to allocate
83     * a buffer intended to hold all data in this stream.
84     *
85     * @return     an estimate of the number of bytes that can be read (or skipped
86     *             over) from this input stream without blocking or {@code 0} when
87     *             it reaches the end of the input stream.
88     * @exception  IOException if an I/O error occurs.
89     */
90    int available() throws IOException;
91  
92    /**
93     * Read up to the specified number of bytes from a given position within a
94     * stream and return the number of bytes read.
95     * This does not change the current offset of the stream and is thread-safe.
96     * 
97     * An implementation may not support positioned read. If the implementation
98     * doesn't support positioned read, it throws UnsupportedOperationException.
99     */
100   int read(long position, byte[] buffer, int offset, int length)
101       throws IOException;
102 
103   /**
104    * Seek to the given offset from the start of the stream.
105    * The next read() will be from that location.
106    * 
107    * An implementation may not support seek. If the implementation 
108    * doesn't support seek, it throws UnsupportedOperationException.
109    */
110   void seek(long position) throws IOException;
111 
112   /**
113    * Closes this input and releases any system resources associated
114    * with the under layer input.
115    *
116    * @exception  IOException  if an I/O error occurs.
117    */
118   void close() throws IOException;
119 }
