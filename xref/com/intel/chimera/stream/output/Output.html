18  package com.intel.chimera.stream.output;
19  
20  import java.io.IOException;
21  import java.nio.ByteBuffer;
22  
23  /**
24   * The Output interface abstract the output target of <code>CryptoOutputStream</code> so that
25   * different implementation of output can be used. The implementation Output interface will usually
26   * wraps an output mechanism such as <code>OutputStream</code> or <code>WritableByteChannel</code>.
27   */
28  
29  public interface Output {
30  
31    /**
32     * Writes a sequence of bytes to this output from the given buffer.
33     *
34     * <p> An attempt is made to write up to <i>r</i> bytes to the channel,
35     * where <i>r</i> is the number of bytes remaining in the buffer, that is,
36     * <tt>src.remaining()</tt>, at the moment this method is invoked.
37     *
38     * <p> Suppose that a byte sequence of length <i>n</i> is written, where
39     * <tt>0</tt>&nbsp;<tt>&lt;=</tt>&nbsp;<i>n</i>&nbsp;<tt>&lt;=</tt>&nbsp;<i>r</i>.
40     * This byte sequence will be transferred from the buffer starting at index
41     * <i>p</i>, where <i>p</i> is the buffer's position at the moment this
42     * method is invoked; the index of the last byte written will be
43     * <i>p</i>&nbsp;<tt>+</tt>&nbsp;<i>n</i>&nbsp;<tt>-</tt>&nbsp;<tt>1</tt>.
44     * Upon return the buffer's position will be equal to
45     * <i>p</i>&nbsp;<tt>+</tt>&nbsp;<i>n</i>; its limit will not have changed.
46     *
47     * @param  src
48     *         The buffer from which bytes are to be retrieved
49     *
50     * @return The number of bytes written, possibly zero
51     *
52     * @throws  IOException
53     *          If some other I/O error occurs
54     */
55    int write(ByteBuffer src) throws IOException;
56    
57    /**
58     * Flushes this output and forces any buffered output bytes
59     * to be written out if the under layer output method support
60     * The general contract of <code>flush</code> is
61     * that calling it is an indication that, if any bytes previously
62     * written have been buffered by the implementation of the output
63     * stream, such bytes should immediately be written to their
64     * intended destination.
65     *
66     * @exception  IOException  if an I/O error occurs.
67     */
68    void flush() throws IOException;
69    
70    /**
71     * Closes this output and releases any system resources associated
72     * with the under layer output.
73     *
74     * @exception  IOException  if an I/O error occurs.
75     */
76    void close() throws IOException;
77  }
